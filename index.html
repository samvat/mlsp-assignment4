def gaussian_pdf(X, mX, Sigma):
    P = np.zeros([2,962])
    for i in range(Sigma.shape[2]):
        C = np.sqrt(((2*np.pi)**12)*np.linalg.det(Sigma[:,:,i]))
        A = np.zeros([12,1])
        for j in range(X.shape[1]):
            A = X[:,j] - mX[:,i]
            B = -0.5*np.dot(A.T, np.dot(np.linalg.inv(Sigma[:,:,i]),A))
            P[i,j] = np.exp(B)/C
    return P

    for i in range(2):

    #calcualting post probabilities
    for i in range(2):

#Power Iteration Routine
def powerIteration(X):
    v_i = np.random.rand(X.shape[1],1)
    for i in range(5000):
        v_i = np.dot(X,v_i)
        v_i /= np.linalg.norm(v_i)
    return v_i
